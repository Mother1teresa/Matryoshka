import axios from "axios";
import { notify } from "/src/utils/notify.js";

export const authApi = axios.create({
  baseURL: "/api",
  withCredentials: true,
});
export const api = axios.create({
  baseURL: "/api",
  withCredentials: true,
});

let isRefreshing = false;
let failedQueue = [];
let refreshAttempts = 0;
const MAX_REFRESH_ATTEMPTS = 3;

const processQueue = (error) => {
  failedQueue.forEach((prom) => {
    if (error) prom.reject(error);
    else prom.resolve();
  });
  failedQueue = [];
};

// ОБРАБОТКА REFRESH КУК
api.interceptors.response.use(
  (response) => response,
  async (error) => {
    const { useAuthStore } = await import("/src/stores/authStore.js");
    const auth = useAuthStore();
    const originalRequest = error.config;
    if (!originalRequest) return Promise.reject(error);
    
    const status = error.response?.status;
    const data = error.response?.data;
    const url = originalRequest.url || "";
    const method = originalRequest.method?.toUpperCase() || "";

    // Защита: эндпоинты авторизации не должны обрабатываться этим интерцептором
    if (url.includes("/auth/")) {
      return Promise.reject(error);
    }

    // 1. Проверка на явный флаг истечения сессии от бэкенда
    if (data?.code === "SESSION_EXPIRED") {
      auth.logout();
      notify("Сессия истекла. Войдите заново.", "error");
      return Promise.reject(error);
    }

    // Извлекаем текст ошибки, если он есть
    const errorMessage = data?.message || data?.error || (typeof data === 'string' ? data : "");

    // 2. ГЛОБАЛЬНАЯ УНИВЕРСАЛЬНАЯ ПРОВЕРКА ДЛЯ ИЗМЕНЕНИЙ И АНОНИМОВ
    // Срабатывает если:
    // - Бэк сообщил, что юзер не найден (актуально при удалении из БД)
    // - Происходит изменение данных (POST, PUT, PATCH, DELETE) и бэк вернул 401/403/404 (юзер стал гостем/анонимом)
    const isMutation = ["POST", "PUT", "PATCH", "DELETE"].includes(method);
    const isAuthError = [401, 403, 404].includes(status);

    if (errorMessage.includes("User not found") || (isMutation && isAuthError)) {
      console.warn(`⛔ Безопасный логаут. Метод: ${method}, Статус: ${status}, Текст: ${errorMessage}`);
      auth.logout();
      notify("Сессия недействительна. Войдите заново.", "error");
      return Promise.reject(error);
    }

    // 3. Стандартная обработка 401 для обычных GET запросов контента (запускает рефреш токенов)
    if (status !== 401) {
      return Promise.reject(error);
    }

    if (originalRequest._retry) {
      console.warn("⛔ Повторный запрос после рефреша вернул ошибку авторизации. Логаут.");
      auth.logout();
      notify("Ошибка авторизации. Войдите заново.", "error");
      return Promise.reject(error);
    }

    if (refreshAttempts >= MAX_REFRESH_ATTEMPTS) {
      console.warn("⛔ Слишком много попыток рефреша кук. Сессия протухла.");
      refreshAttempts = 0;
      auth.logout();
      notify("Сессия истекла. Войдите заново.", "error");
      return Promise.reject(error);
    }

    if (isRefreshing) {
      return new Promise((resolve, reject) => {
        failedQueue.push({ resolve, reject });
      })
        .then(() => {
          originalRequest._retry = true;
          return api(originalRequest);
        })
        .catch((err) => {
          auth.logout();
          return Promise.reject(err);
        });
    }

    originalRequest._retry = true;
    isRefreshing = true;
    refreshAttempts++;

    try {
      const success = await auth.refreshToken();
      if (success) {
        refreshAttempts = 0;
        await new Promise(resolve => setTimeout(resolve, 50));
        processQueue(null);
        return api(originalRequest);
      }
      throw new Error("Refresh failed");
    } catch (refreshError) {
      processQueue(refreshError);
      auth.logout();
      notify("Сессия истекла. Войдите заново.", "error");
      return Promise.reject(refreshError);
    } finally {
      isRefreshing = false;
    }
  },
);

export const resetRefreshCooldown = () => {
  refreshAttempts = 0;
};
